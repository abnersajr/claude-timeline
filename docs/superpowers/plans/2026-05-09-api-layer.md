# API Layer Implementation Plan

> **For Hermes:** Use subagent-driven-development skill to implement this plan task-by-task.

**Goal:** Add an HTTP API server (`@timeline/api`) that wraps the extractor as a data layer, serving session data to the webapp via JSON endpoints.

**Architecture:** Simple layered Express server in a new `api/` workspace package. Zod for runtime validation + TypeScript type inference. Hand-written OpenAPI spec for documentation + webapp codegen. Imports `@timeline/extractor` submodules directly (no barrel files). Local dev uses `localias` for HTTPS custom domains + Turborepo for concurrent dev servers.

**Tech Stack:** Express, Zod V4, TypeScript, Biome, Turborepo, localias

**Step 2: Create `api/src/config.ts`**

```typescript
import { readFileSync } from "node:fs"
import { resolve } from "node:path"
import { getDbPath, getProjectsDir } from "@timeline/extractor/utils"

export interface ApiConfig {
  port: number
  dbPath: string
  projectsDir: string
  corsOrigins: string[]
}

interface ConfigFile {
  port?: number
  corsOrigins?: string[]
  dbPath?: string | null
  projectsDir?: string | null
}

/**
 * Load API configuration.
 * Priority: env vars > config.json > extractor defaults
 */
export function loadConfig(configDir?: string): ApiConfig {
  const baseDir = configDir ?? process.cwd()
  const filePath = resolve(baseDir, "config.json")

  let file: ConfigFile = {}
  try {
    file = JSON.parse(readFileSync(filePath, "utf-8"))
  } catch {
    // config.json is optional — fall back to env/defaults
  }

  const port = Number.parseInt(process.env.PORT ?? String(file.port ?? 3001), 10)

  const corsOrigins = process.env.CORS_ORIGIN
    ? process.env.CORS_ORIGIN.split(",").map((s) => s.trim())
    : file.corsOrigins ?? ["https://claude-dash.local"]

  const dbPath = getDbPath(process.env.CLAUDE_DB_PATH ?? file.dbPath ?? undefined)
  const projectsDir = getProjectsDir(process.env.CLAUDE_PROJECTS_DIR ?? file.projectsDir ?? undefined)

  return { port, dbPath, projectsDir, corsOrigins }
}
```

## Task 3: In-memory session cache

**Objective:** Cache extracted session data in memory, invalidating when the source files change.

**Files:**
- Create: `api/src/cache.ts`

**Step 1: Create `api/src/cache.ts`**

```typescript
import { statSync } from "node:fs"
import type { FullTimelineSession } from "@timeline/extractor/types"

interface CacheEntry {
  data: FullTimelineSession
  dbMtime: number
}

/**
 * In-memory cache for extracted sessions.
 * Invalidates when the SQLite DB mtime changes.
 */
export class SessionCache {
  private cache = new Map<string, CacheEntry>()

  get(sessionId: string, dbPath: string): FullTimelineSession | null {
    const entry = this.cache.get(sessionId)
    if (!entry) return null

    const currentMtime = getMtime(dbPath)
    if (currentMtime > entry.dbMtime) {
      this.cache.delete(sessionId)
      return null
    }

    return entry.data
  }

  set(sessionId: string, data: FullTimelineSession, dbPath: string): void {
    this.cache.set(sessionId, {
      data,
      dbMtime: getMtime(dbPath),
    })
  }

  clear(): void {
    this.cache.clear()
  }
}

function getMtime(path: string): number {
  try {
    return statSync(path).mtimeMs
  } catch {
    return 0
  }
}
```

**Step 2: Create `api/src/routes/sessions.ts`**

```typescript
import { Router } from "express"
import { z } from "zod/v4"
import { extractFullTimeline } from "@timeline/extractor/merger"
import { listSessions } from "@timeline/extractor/db-reader"
import { SessionCache } from "../cache.js"
import type { ApiConfig } from "../config.js"

const cache = new SessionCache()

const ListSessionsQuerySchema = z.object({
  limit: z.coerce.number().int().positive().max(100).optional().default(20),
})

const SessionIdParamSchema = z.object({
  id: z.string().min(1),
})

export function createSessionsRouter(config: ApiConfig) {
  const router = Router()

  /**
   * GET /api/sessions
   * List recent sessions ordered by most recent first.
   */
  router.get("/sessions", (req, res) => {
    const parsed = ListSessionsQuerySchema.safeParse({ limit: req.query.limit })
    if (!parsed.success) {
      res.status(400).json({ error: "Invalid query parameters", details: parsed.error.message })
      return
    }

    try {
      const sessions = listSessions(config.dbPath, parsed.data.limit)
      res.json(sessions)
    } catch (err) {
      res.status(500).json({ error: "Failed to list sessions", details: String(err) })
    }
  })

  /**
   * GET /api/sessions/:id
   * Full session timeline with all fields.
   */
  router.get("/sessions/:id", async (req, res) => {
    const parsed = SessionIdParamSchema.safeParse({ id: req.params.id })
    if (!parsed.success) {
      res.status(400).json({ error: "Invalid session ID" })
      return
    }

    const sessionId = parsed.data.id

    // Check cache first
    const cached = cache.get(sessionId, config.dbPath)
    if (cached) {
      res.json(cached)
      return
    }

    try {
      const data = await extractFullTimeline(sessionId, config.dbPath, config.projectsDir)
      cache.set(sessionId, data, config.dbPath)
      res.json(data)
    } catch (err) {
      const message = String(err)
      if (message.includes("Session not found")) {
        res.status(404).json({ error: "Session not found", sessionId })
        return
      }
      res.status(500).json({ error: "Failed to extract session", details: message })
    }
  })

  return router
}
```

    const res = await request(app).get(`/api/sessions/${sessions[0].sessionId}`)

    expect(res.body).toHaveProperty("session")
    expect(res.body).toHaveProperty("turns")
    expect(res.body).toHaveProperty("pricing")
    expect(res.body.session.sessionId).toBe(sessions[0].sessionId)
  })

  it("serves cached response on second request", async () => {
    const listRes = await request(app).get("/api/sessions?limit=1")
    const sessions = listRes.body
    if (sessions.length === 0) return

    const id = sessions[0].sessionId
    const res1 = await request(app).get(`/api/sessions/${id}`)
    const res2 = await request(app).get(`/api/sessions/${id}`)

        model:

        commandExecuted:

        workingDirectory:

        turnCount:

          format: date-time

            description: Turn data (detailed schema deferred)
